	  var cTemp = celsius;
	  var cToFahr = cTemp * 9 / 5 + 32;

	function fToC(fahrenheit) {
	  var fTemp = fahrenheit;

                    if (n%2!=0) {
                        return false;
                    }

                    var result = 0,
                    x = 1;
                    while (n != 0) {
                        var dg1 = n % 10, dg2 = ((n - dg1) / 10) % 10;

                        x *= 100;

                }
